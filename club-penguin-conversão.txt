import java.util.Scanner;

public class HelloWorld {
    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);

        System.out.print("Quantas moedas você tem? ");
        int moedas = input.nextInt();

        double taxaConversao =  0.0045;

        double total = moedas * taxaConversao;

        System.out.println("Você tem " + total + " em USD");

        input.close();
        
    }
}
